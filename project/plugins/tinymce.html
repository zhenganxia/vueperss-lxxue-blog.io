  },

      type: String,
      default: ''
    },

      type: Boolean,

    }
  }

# 组件：XdTinymce

<template>
    <editor v-model="myValue" :init="init" :disabled="disabled" :id="tinymceId" ref="tinymceRef"></editor>
</template>
  
<script>
export default {
    name: 'XdTinymce'
}
</script>
<script setup>
//在js中引入所需的主题和组件
import tinymce from 'tinymce/tinymce'
import 'tinymce/skins/content/default/content.css'
import Editor from '@tinymce/tinymce-vue'
import 'tinymce/themes/silver'
import 'tinymce/themes/silver/theme'
//在TinyMce.vue中接着引入相关插件-展示需要处理下props.plugins参数
import 'tinymce/icons/default'; //引入编辑器图标icon，不引入则不显示对应图标
import 'tinymce/models/dom'
import "tinymce/icons/default/icons"
import "tinymce/plugins/image" // 插入上传图片插件
import "tinymce/plugins/media" // 插入视频插件
import "tinymce/plugins/table" // 插入表格插件
import "tinymce/plugins/lists" // 列表插件
import "tinymce/plugins/wordcount" // 字数统计插件
import "tinymce/plugins/code" // 源码
import "tinymce/plugins/fullscreen" //全屏

import { reactive, ref } from "vue"
import { onMounted, defineEmits, watch } from "@vue/runtime-core"
import { ElMessage } from "element-plus";
const emits = defineEmits(["update:modelValue","on-validate"])
const props = defineProps({
    modelValue: {
        type: String,
        default: () => {
            return ""
        },
    },
    baseUrl: {
        type: String,
        default: "",
    },
    disabled: {
        type: Boolean,
        default: false,
    },
    plugins: {
        type: [String, Array],
        default: "lists  table  image  wordcount ", // 限制字数添加ax_wordlimit 和maxlength配合使用
    },
    toolbar: {
        type: [String, Array],
        default:
            "codesample bold italic underline alignleft aligncenter alignright alignjustify | undo redo | formatselect | fontselect | fontsizeselect | forecolor backcolor | bullist numlist outdent indent | lists link table code | removeformat ",
    },
    maxSize: {
        type: Number,
        default: 2
    },
    height:{
        type: Number,
        default: 400
    },
    maxlength:{
        type: Number,
        default: 800
    }
})

const myValue = ref()
const tinymceRef = ref()
const tinymceId = ref("vue-tinymce-" + +new Date() + ((Math.random() * 1000).toFixed(0) + ""))
//初始化
const init = reactive({
    selector: '#' + tinymceId.value, //富文本编辑器的id,
    language_url: './tinymce/langs/zh-Hans.js',
    language: "zh-Hans", //语言
    skin_url: "./tinymce/skins/ui/oxide", // skin路径
    height: props.height, //编辑器高度
    branding: false, //是否禁用“Powered by TinyMCE”
    menubar: true, //顶部菜单栏显示
    image_dimensions: false, //去除宽高属性
    plugins: props.plugins,  // 这里有默认值-支持手动传参覆盖
    toolbar: props.toolbar, //这里的数据是在props里面就定义好了的
    font_formats: 'Arial=arial,helvetica,sans-serif; 宋体=SimSun; 微软雅黑=Microsoft Yahei; Impact=impact,chicago;', //字体
    fontsize_formats: '11px 12px 14px 16px 18px 24px 36px 48px 64px 72px', // 文字大小
    // paste_convert_word_fake_lists: false, // 插入word文档需要该属性
    paste_webkit_styles: "all",
    external_plugins: {  // 限制字数插件
        ax_wordlimit: './tinymce/plugins/ax_wordlimit/plugin.min.js'
    },
    ax_wordlimit_num: props.maxlength, // 限制字数
    ax_wordlimit_callback: function (editor, txt, num,validate) { // 限制字数回调
        console.log('当前字数：' + txt + '，限制字数：' + num,validate);
        ElMessage({
        type: "warning",
        message: `限制字数${num},请删除多余字数`
      });
    },
    paste_merge_formats: true,
    nonbreaking_force_tab: false,
    paste_auto_cleanup_on_paste: false,
    file_picker_types: 'file',
    content_css: './tinymce/skins/content/default/content.css', //以css文件方式自定义可编辑区域的css样式，css文件需自己创建并引入
    //图片上传
    images_upload_handler: (blobInfo) => new Promise((resolve, reject) => {
        if (blobInfo.blob().size / 1024 / 1024 > props.maxSize) {
            reject({ message: `上传失败，图片大小请控制在 ${props.maxSize}M以内`, remove: true })
            return
        } else {
            const img = 'data:image/jpeg;base64,' + blobInfo.base64()
            resolve(img)
        }
    }),
    // 文件上传
    file_picker_callback: (callback, value, meta) => {
        // Provide file and text for the link dialog
        if (meta.filetype == 'file') {
            callback('mypage.html', { text: 'My text' });
        }

        // Provide image and alt text for the image dialog
        if (meta.filetype == 'image') {
            callback('myimage.jpg', { alt: 'My alt text' });
        }

        // Provide alternative source and posted for the media dialog
        if (meta.filetype == 'media') {
            callback('movie.mp4', { source2: 'alt.ogg', poster: 'image.jpg' });
        }
    },
    init_instance_callback() {
        // 解决字数统计-wordcount只统计了中文字的字数，数字和字母被忽略问题 editor.getContainer()可以获取所有div
        if(props.plugins.indexOf('wordcount') > 0 ) {
            const lastDom = document.querySelector('button.tox-statusbar__wordcount');
            lastDom.click()
        }
    },
})

//监听外部传递进来的的数据变化
watch(
    () => props.modelValue,
    () => {
        myValue.value = props.modelValue
    },
    {
        immediate:true
    }
)
//监听富文本中的数据变化
watch(
    () => myValue.value,
    () => {
        emits("update:modelValue", myValue.value)
    }
)

//在onMounted中初始化编辑器
onMounted(() => {
    tinymce.init({})
})
</script>
